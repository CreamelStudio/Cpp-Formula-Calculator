long double Calculate(long double a, long double b, char oper){
    switch(oper){
        case '+':
            return a + b;
            break;
        case '-':
            return a - b;
            break;
        case '*':
            return a * b;
            break;
        case '/':
            return a / b;
            break;
        default:
            return -100000000000;
            break;
    }
}
